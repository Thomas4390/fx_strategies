\section{Résumé exécutif}
\label{sec:executive_summary}

\lettrine[lines=2, findent=2pt, nindent=0pt]{L}{a} stratégie Phase~18 combine \textbf{trois moteurs de rendement orthogonaux}~--- un \emph{mean reversion} intraday filtré par un régime macroéconomique, un \emph{momentum} daily sur trois paires FX majeures, et un \emph{mean reversion} daily sur l'indicateur RSI à quatre paires~--- le tout piloté par une couche de vol-targeting global capée à 12$\times$. C'est l'aboutissement d'une série d'itérations (Phases~14 à~18) documentée dans le journal de bord interne.

\subsection*{Les quatre observations critiques}

\begin{enumerate}
\item \textbf{OOS Sharpe 1.44 $>$ IS Sharpe 0.94.} L'amélioration hors sample valide la robustesse de la structure multi-sleeves.
\item \textbf{Bootstrap P5 Max Drawdown~$-30.68\,\%$} strictement sous le cap 35\,\%, avec environ 4~pp de marge de sécurité.
\item \textbf{Six années positives sur sept} en walk-forward~--- seule 2019 reste marginalement négative (Sharpe $-0.49$, régime \emph{late-cycle range-bound}).
\item \textbf{Première configuration} de la série Phases~14 à~18 qui franchit \emph{simultanément} les gates in-sample, out-of-sample et bootstrap sans caveat méthodologique.
\end{enumerate}
